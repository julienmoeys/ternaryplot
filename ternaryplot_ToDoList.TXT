
ADD argument 'classes' to ternaryPlot()

ADD package argument 'class.label.cex' and 'class.label.font' 
    in tpPar() and use them in ternaryPolygons.ternarySystem()

ADD argument 'grid' to ternaryPlot()

ADD argument 'type' = 't' (text) to ternaryPlot()

CREATE ternaryCounts ternarySystem add type = "c"
    to ternaryPlot().

CREATE ternaryLines ternarySystem and add type = "l"
    to ternaryPlot().

CREATE ternaryClassNames() and ternaryClassNames()<-
    methods to retrieve and change the (abbrev) name of 
    ternary classes

CREATE as.data.frame.ternarySystem method
    what = "vertices" or "classes"

CREATE ternaryAggregate >> ternary version of sp::aggregate

CREATE plot method for ternarySystem and ternaryPolygons
